const tabBtn = $(".tab-btn ul li");
    const tabCtn = $(".tab-cont > div");

    tabBtn.click(function(){
        let target = $(this);
        tabBtn.removeClass("active");
        target.addClass("active");

        let index = target.index();
        tabCtn.css("display","none");
        tabCtn.eq(index).css("display","block");
    });
